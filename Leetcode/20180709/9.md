class Solution {
    public boolean isPalindrome(int x) {
        String s1=Integer.toString(x);
        StringBuffer s=new StringBuffer(s1).reverse();
        String s2=s.toString();
        if(s1.equals(s2)){
            System.out.println("true");
            return true;
        } 
        else {
            System.out.println("false");
            return false;
        }
    }
}
