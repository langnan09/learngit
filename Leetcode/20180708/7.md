class Solution {
    public int reverse(int x) {
        long temp=x;
        long num=0;
        do{
            int a;
            a=(int)temp%10;
            num=num*10+a;
            temp=temp/10;
        }while(temp!=0);
        if(num<Integer.MIN_VALUE||num> Integer.MAX_VALUE) return 0;
        return  (int)num;
    }
}
